package main

import (
	"fmt"
	"regexp"
)

func main() {
	pattern := "go(lang)?"

	regex, err := regexp.Compile(pattern)

	if err != nil {
		fmt.Println("Error compiling regular expression: ", err)
		return
	}

	testString := "love go programming language!"

	if regex.MatchString(testString) {
		fmt.Println("The test string mactes")
	} else {
		fmt.Println("the test string doesn't match")
	}

}
